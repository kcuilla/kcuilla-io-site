2019 NFL Team Ratings

Aug 15, 2020 14 min read R

2019 NFL Team Stats & Ratings

Created by: @kc_analytics

Source: Pro-Football-Reference

Process

The main inspiration for this post came from 538’s NBA Player Ratings which shows the player ratings in an interactive chart and table that are linked together. My goal was to build a similar-styled data visualization for NFL team ratings from the 2019 season and make it reproducible so that it could be easily updated for the upcoming 2020 season.

A couple great tutorials that I used to create the interactive table from the reactable package were from Greg Lin, who wrote one of the most comprehensive tutorials I’ve seen out there for an R package, and also Tom Mock, who wrote a fantastic tutorial that really helped me understand how to use CSS and Google fonts to enhance the visual appeal of the table.

For creating the interactive plot, I frequently referenced this Plotly guide from the makers of the plotly package.

And for linking the reactable table and plotly plot together, I used the crosstalk package which can be found here.

Data

The data I used comes from the 2019 NFL Standings & Team Stats page on the Pro Football Reference website. I utilized the rvest package to scrape the data from the AFC and NFC Standings table and combined them into a single table.

I then used packages such as dplyr, tidyr, and stringr to clean and tidy the data so that it could be easily used for visualization.

Visualization

For the Strength-of-Schedule (SoS) metric from Pro-Football-Reference, I thought it would be more easily understood if I bucketed the metric into groups and assigned an easy-to-difficult scale. I based my scale off of the ski trail difficulty scale below:

I also decided to remove the asterisks that are present in the original Pro-Football-Reference table that signify if the team made the playoffs or not and converted this into a new column with a green checkmark for teams that made the playoffs and a red x for teams that didn’t so that we could filter this column for playoff teams vs non-playoff teams if desired.

I also made the plot above the table interactive so that when you hover over a point you can see the team name and the OSRS and DSRS rating of that team. You can also click-and-drag over a group of data points to display just those data points in the table and double-click on the plot to de-select.

Both the plot and the table are controlled by the filters that are located on the left-hand side of the plot.

bar_chart <-
  function(label,
           width = "100%",
           height = "13px",
           fill = "#00bfc4",
           background = NULL) {
    bar <-
      div(style = list(
        background = fill,
        width = width,
        height = height
      ))
    chart <-
      div(style = list(
        flexGrow = 1,
        marginLeft = "8px",
        background = background
      ),
      bar)
    div(style = list(display = "flex", alignItems = "center"), label, chart)
  }

make_color_pal <- function(colors, bias = 1) {
  get_color <- colorRamp(colors, bias = bias)
  function(x)
    rgb(get_color(x), maxColorValue = 255)
}

off_rating_color <-
  make_color_pal(c("#67a9cf", "#f8fcf8", "#ef8a62"), bias = 1.3)
def_rating_color <-
  make_color_pal(c("#67a9cf", "#f8fcf8", "#ef8a62"), bias = 0.8)

team_rating_column <- function(maxWidth = 60, ...) {
  colDef(
    maxWidth = maxWidth,
    align = "right",
    class = "cell number",
    headerStyle = list(fontWeight = "500"),
    ...
  )
}

conf_filter <- filter_checkbox(
  id = "conf",
  label = "Conference",
  inline = TRUE,
  sharedData = NFL_table_clean_shared,
  group = ~ Conference
)

div_filter <- filter_select(
  id = "divi",
  label = "Division",
  sharedData = NFL_table_clean_shared,
  group = ~ Division
)

sos_filter <- filter_checkbox(
  id = "sos",
  label = "Strength of Schedule (SoS)",
  sharedData = NFL_table_clean_shared,
  group = ~ SoS_rating
)

playoff_filter <- filter_checkbox(
  id = "play",
  label = "Playoffs",
  inline = TRUE,
  sharedData = NFL_table_clean_shared,
  group = ~ Playoffs
)

srs_filter <- filter_slider(
  id = "srs",
  label = "Team Rating (SRS): Total",
  sharedData = NFL_table_clean_shared,
  column = ~ SRS,
  ticks = FALSE,
  step = 1
)

table <-
  reactable(
    NFL_table_clean_shared,
    pagination = FALSE,
    showSortIcon = FALSE,
    highlight = TRUE,
    compact = TRUE,
    defaultSorted = "SRS",
    defaultSortOrder = "desc",
    defaultColDef = colDef(headerClass = "header colheader"),
    columnGroups = list(
      colGroup(
        name = "Team Rating (SRS)",
        columns = c("SRS", "OSRS", "DSRS"),
        headerClass = "groupheader"
      ),
      colGroup(
        name = "Team Scoring & Margin of Victory",
        columns = c("PF", "PA", "MoV"),
        headerClass = "groupheader"
      )
    ),
    columns = list(
      Division = colDef(show = FALSE),
      Conference = colDef(show = FALSE),
      Rank = colDef(
        name = "#",
        maxWidth = 30,
        class = "cell number"
      ),
      Team = colDef(
        minWidth = 150,
        class = "cell",
        cell = function(value, index) {
          div(div(class = "team-name", value),
              div(class = "record", sprintf("(%s)", NFL_table_clean[index, "Record"])))
        }
      ),
      Record = colDef(show = FALSE),
      SRS = team_rating_column(
        name = "Total",
        cell = function(value) {
          normalized <-
            (value - min(NFL_table_clean$SRS)) / (max(NFL_table_clean$SRS) - min(NFL_table_clean$SRS))
          color <- off_rating_color(normalized)
          value <- format(value, nsmall = 1, digits = 1)
          div(class = "roundcorners",
              style = list(background = color),
              value)
        }
      ),
      OSRS = team_rating_column(
        name = "Off.",
        cell = function(value) {
          normalized <-
            (value - min(NFL_table_clean$SRS)) / (max(NFL_table_clean$SRS) - min(NFL_table_clean$SRS))
          color <- off_rating_color(normalized)
          value <- format(value, nsmall = 1, digits = 1)
          div(class = "roundcorners",
              style = list(background = color),
              value)
        }
      ),
      DSRS = team_rating_column(
        name = "Def.",
        cell = function(value) {
          normalized <-
            (value - min(NFL_table_clean$SRS)) / (max(NFL_table_clean$SRS) - min(NFL_table_clean$SRS))
          color <- off_rating_color(normalized)
          value <- format(value, nsmall = 1, digits = 1)
          div(class = "roundcorners",
              style = list(background = color),
              value)
        }
      ),
      PF = colDef(
        name = "Points Scored",
        align = "left",
        class = "border-left cell number",
        headerStyle = list(fontWeight = "500"),
        cell = function(value) {
          width <- paste0(value / max(NFL_table_clean$PF) * 100, "%")
          bar_chart(value,
                    width = width,
                    fill = "#ef8a62",
                    background = "#e1e1e1")
        }
      ),
      PA = colDef(
        name = "Points Against",
        align = "left",
        class = "cell number",
        headerStyle = list(fontWeight = "500"),
        cell = function(value) {
          width <- paste0(value / max(NFL_table_clean$PA) * 100, "%")
          bar_chart(value,
                    width = width,
                    fill = "#ef8a62",
                    background = "#e1e1e1")
        }
      ),
      PD = colDef(show = FALSE),
      MoV = colDef(
        maxWidth = 60,
        class = "border-right cell number",
        headerStyle = list(fontWeight = "500"),
        cell = function(value) {
          if (value > 0)
            paste0("+", value)
          else
            value
        },
        style = function(value) {
          if (value > 0) {
            color <- "#008000"
          } else if (value < 0) {
            color <- "#e00000"
          } else {
            color <- "#777"
          }
          list(color = color)
        }
      ),
      SoS_rating = colDef(
        name = "SoS",
        align = "center",
        maxWidth = 55,
        class = "border-left cell number",
        cell = function(value) {
          if (value == "Most Difficult") {
            list(tagAppendAttributes(shiny::icon("square", class = "rotate")),
                 tagAppendAttributes(shiny::icon("square", class = "rotate")))
          } else if (value == "Difficult") {
            tagAppendAttributes(shiny::icon("square", class = "rotate"))
          } else if (value == "Moderate") {
            tagAppendAttributes(shiny::icon("square"))
          } else {
            tagAppendAttributes(shiny::icon("circle"))
          }
        },
        style = function(value) {
          if (value %in% c("Difficult", "Most Difficult")) {
            color <- "black"
          } else if (value == "Moderate") {
            color <- "blue"
          } else {
            color <- "green"
          }
          list(color = color)
        }
      ),
      Playoffs = colDef(
        maxWidth = 75,
        align = "center",
        class = "cell number border-left",
        cell = function(value) {
          if (value == "Yes")
            tagAppendAttributes(shiny::icon("check"))
          else
            tagAppendAttributes(shiny::icon("times"))
        },
        style = function(value) {
          if (value == "Yes") {
            color <- "green"
          } else {
            color <- "red"
          }
          list(color = color)
        }
      )
    )
  )
